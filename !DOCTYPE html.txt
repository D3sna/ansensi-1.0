        async function startCamera() {
            const btn = document.getElementById('btn-toggle-camera');

        function handleScan(id) {
            const emp = employees.find(e => e.id === id);
            if (!emp) return showToast("QR Tidak Dikenal!", "error");

            const today = new Date().toLocaleDateString('id-ID');
            const nowTime = new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' }).replace('.', ':');
            let log = logs.find(l => l.id === id && l.date === today);

            if (!log) {
                const isLate = nowTime > sysSettings.masuk;
                logs.unshift({ id, name: emp.name, date: today, timeIn: nowTime, timeOut: null, isLate, isEarly: false });
                showToast(`Halo ${emp.name.split(' ')[0]}, Selamat Bekerja!`, "success");
            } else if (!log.timeOut) {
                const tIn = parseInt(log.timeIn.split(':')[0]) * 60 + parseInt(log.timeIn.split(':')[1]);
                const tNow = parseInt(nowTime.split(':')[0]) * 60 + parseInt(nowTime.split(':')[1]);
                
                if (tNow - tIn < 5) return showToast("Gagal! Anda baru saja memindai.", "warning");

                log.timeOut = nowTime;
                log.isEarly = nowTime < sysSettings.pulang;
                showToast(`Sampai jumpa, ${emp.name.split(' ')[0]}!`, "success");
            } else {
                showToast("Anda sudah absen pulang.", "warning");
            }
            saveAndSync();
        }

                        </div>
                        <div class="${isOvertime ? 'bg-indigo-50/50 border-indigo-100' : 'bg-slate-900 border-slate-800'} p-2.5 rounded-2xl border flex flex-col justify-center">
                            <p class="text-[8px] ${isOvertime ? 'text-indigo-600' : 'text-slate-400'} font-extrabold uppercase tracking-widest mb-1">Durasi</p>
                            <p class="text-sm font-black ${isOvertime ? 'text-indigo-700' : 'text-white'}">${formattedHours}</p>
                        </div>
                    </div>
                </div>
            `}).join('') || `<div class="text-center py-10 bg-slate-50 rounded-3xl border border-dashed border-slate-200 text-slate-400 font-medium"><i data-lucide="inbox" class="w-10 h-10 mx-auto mb-2 opacity-50"></i>Data laporan masih kosong</div>`;
            
            lucide.createIcons();
        }

        function calculateHours(start, end) {
            if (!start || !end) return 0;
            const s = start.split(':');
            const e = end.split(':');
            const diff = (parseInt(e[0])*60 + parseInt(e[1])) - (parseInt(s[0])*60 + parseInt(s[1]));
            return diff / 60;
        }

        function openEdit(index) {
            currentEditIdx = index;
            const log = logs[index];
            document.getElementById('edit-in').value = log.timeIn.replace('.', ':');
            document.getElementById('edit-out').value = log.timeOut ? log.timeOut.replace('.', ':') : '';
            document.getElementById('edit-modal').classList.remove('hidden');
        }

        function closeEditModal() { document.getElementById('edit-modal').classList.add('hidden'); }

        function saveEdit() {
            const log = logs[currentEditIdx];
            log.timeIn = document.getElementById('edit-in').value;
            log.timeOut = document.getElementById('edit-out').value || null;
            
            log.isLate = log.timeIn > sysSettings.masuk;
            log.isEarly = log.timeOut ? log.timeOut < sysSettings.pulang : false;
            
            saveAndSync();
            closeEditModal();
            showToast("Log diperbarui!", "success");
        }

        function deleteLog() {
            if(confirm('Hapus log permanen?')) {
                logs.splice(currentEditIdx, 1);
                saveAndSync();
                closeEditModal();
                showToast("Data dibersihkan", "info");
            }
        }

        function saveAndSync() {
            localStorage.setItem('absensi_logs', JSON.stringify(logs));
            renderLogs();
        }

        function showToast(msg, type) {
            const toast = document.getElementById('toast');
            const bg = document.getElementById('toast-bg');
            const txt = document.getElementById('toast-text');
            const icon = document.getElementById('toast-icon');

            txt.innerText = msg;
            
            toast.classList.remove('hidden');
            setTimeout(() => {
                toast.classList.remove('scale-95', 'opacity-0');
                toast.classList.add('scale-100', 'opacity-100');
            }, 10);
            
            if (type === 'success') {
                bg.className = "px-5 py-3.5 rounded-2xl shadow-2xl flex items-center gap-3 text-white font-bold border border-emerald-400/30 bg-gradient-to-r from-emerald-500 to-teal-500";
                icon.innerHTML = '<i data-lucide="check-circle-2" class="w-5 h-5"></i>';
            } else if (type === 'error') {
                bg.className = "px-5 py-3.5 rounded-2xl shadow-2xl flex items-center gap-3 text-white font-bold border border-rose-400/30 bg-gradient-to-r from-rose-500 to-red-500";
                icon.innerHTML = '<i data-lucide="alert-circle" class="w-5 h-5"></i>';
            } else {
                bg.className = "px-5 py-3.5 rounded-2xl shadow-2xl flex items-center gap-3 text-white font-bold border border-amber-400/30 bg-gradient-to-r from-amber-500 to-orange-500";
                icon.innerHTML = '<i data-lucide="info" class="w-5 h-5"></i>';
            }

            lucide.createIcons();
            
            setTimeout(() => {
                toast.classList.remove('scale-100', 'opacity-100');
                toast.classList.add('scale-95', 'opacity-0');
                setTimeout(() => toast.classList.add('hidden'), 300);
            }, 3000);
        }

        function downloadCSV() {
            let csv = "ID,Nama,Tanggal,Masuk,Keluar,Total Jam,Terlambat,Pulang Cepat,Lembur,Status\n";
            logs.forEach(l => {
                const hours = calculateHours(l.timeIn, l.timeOut);
                const isOvertime = hours > 8;
                const totalJam = l.timeOut ? hours.toFixed(2) : '-';
                const status = isOvertime ? "Lembur" : (l.timeOut ? "Hadir Normal" : "Sedang Bekerja");
                
                csv += `${l.id},${l.name},${l.date},${l.timeIn},${l.timeOut || '-'},${totalJam},${l.isLate ? 'YA' : 'TIDAK'},${l.isEarly ? 'YA' : 'TIDAK'},${isOvertime ? 'YA' : 'TIDAK'},${status}\n`;
            });
            
            const blob = new Blob([csv], { type: 'text/csv' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `Rekap_Presensi_${new Date().toLocaleDateString('id-ID').replace(/\//g,'-')}.csv`;
            a.click();
        }
