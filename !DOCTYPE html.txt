    <script>
        // State Management
        let logs = JSON.parse(localStorage.getItem('absensi_logs')) || [];
        let employees = JSON.parse(localStorage.getItem('absensi_emps')) || [
            { id: 'ADM01', name: 'Admin Utama' }
        ];
        let currentEditIdx = null;
        let isScannerActive = false;
        let html5QrCode = null;

        const JAM_MASUK_MAKS = "08:00";

        // Inisialisasi
        window.onload = () => {
            renderEmployees();
            renderLogs();
            lucide.createIcons();
        };

        // Navigation Logic
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
            document.getElementById(`tab-${tabId}`).classList.add('active');
            
            // Update Nav Icons
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.replace('text-blue-600', 'text-slate-400');
            });
            document.querySelector(`[data-tab="${tabId}"]`).classList.replace('text-slate-400', 'text-blue-600');

            // Update Header
            const title = document.getElementById('page-title');
            const sub = document.getElementById('page-subtitle');
            const role = document.getElementById('role-indicator');

            if (tabId === 'scan') {
                title.innerText = "PresensiKita";
                sub.innerText = "Scan QR untuk masuk/keluar";
                role.innerText = "User Mode";
                role.className = "bg-blue-600 text-white text-[10px] font-bold px-3 py-1 rounded-full uppercase tracking-widest";
            } else {
                title.innerText = tabId === 'employees' ? "Data Karyawan" : "Laporan Admin";
                sub.innerText = "Manajemen data & kehadiran";
                role.innerText = "Admin Mode";
                role.className = "bg-amber-500 text-white text-[10px] font-bold px-3 py-1 rounded-full uppercase tracking-widest";
            }
            lucide.createIcons();
        }

        // --- MANAJEMEN KARYAWAN ---
        function addEmployee() {
            const name = document.getElementById('emp-name').value;
            const id = document.getElementById('emp-id').value.toUpperCase().replace(/\s/g, '');

            if (!name || !id) return showToast("Lengkapi data!", "error");
            if (employees.find(e => e.id === id)) return showToast("ID sudah ada!", "error");

            employees.push({ id, name });
            localStorage.setItem('absensi_emps', JSON.stringify(employees));
            
            document.getElementById('emp-name').value = '';
            document.getElementById('emp-id').value = '';
            
            renderEmployees();
            showToast("Karyawan ditambahkan", "success");
        }

        function renderEmployees() {
            const list = document.getElementById('employee-list');
            list.innerHTML = employees.map(emp => `
                <div class="bg-white p-4 rounded-2xl border border-slate-200 flex items-center gap-4">
                    <div id="qr-${emp.id}" class="bg-white p-1 border rounded-lg shrink-0"></div>
                    <div class="flex-1">
                        <h4 class="font-bold text-slate-800">${emp.name}</h4>
                        <p class="text-xs text-slate-500 font-mono">${emp.id}</p>
                        <div class="flex gap-3 mt-2">
                            <button onclick="downloadQR('${emp.id}', '${emp.name}')" class="text-[10px] text-blue-600 font-bold flex items-center gap-1">
                                <i data-lucide="download" class="w-3 h-3"></i> UNDUH QR
                            </button>
                            <button onclick="removeEmployee('${emp.id}')" class="text-[10px] text-red-400 font-bold">HAPUS</button>
                        </div>
                    </div>
                </div>
            `).join('');

            // Generate QR Codes
            employees.forEach(emp => {
                new QRCode(document.getElementById(`qr-${emp.id}`), {
                    text: emp.id,
                    width: 60,
                    height: 60,
                    colorDark : "#0f172a",
                    correctLevel : QRCode.CorrectLevel.H
                });
            });
            lucide.createIcons();
        }

        function removeEmployee(id) {
            if(id === 'ADM01') return;
            employees = employees.filter(e => e.id !== id);
            localStorage.setItem('absensi_emps', JSON.stringify(employees));
            renderEmployees();
        }

        function downloadQR(id, name) {
            const canvas = document.querySelector(`#qr-${id} canvas`);
            const link = document.createElement('a');
            link.download = `QR_${name}_${id}.png`;
            link.href = canvas.toDataURL();
            link.click();
        }

        // --- SCANNER LOGIC ---
        document.getElementById('btn-toggle-camera').onclick = async () => {
            const btn = document.getElementById('btn-toggle-camera');
            const placeholder = document.getElementById('scan-placeholder');

            if (!isScannerActive) {
                html5QrCode = new Html5Qrcode("reader");
                try {
                    placeholder.classList.add('hidden');
                    await html5QrCode.start({ facingMode: "environment" }, { fps: 10, qrbox: 250 }, (code) => {
                        handleScan(code);
                        html5QrCode.pause();
                        setTimeout(() => html5QrCode.resume(), 3000);
                    });
                    isScannerActive = true;
                    btn.innerText = "Matikan";
                    btn.classList.add('bg-red-50', 'text-red-600');
                } catch (e) { showToast("Kamera error", "error"); }
            } else {
                await html5QrCode.stop();
                isScannerActive = false;
                placeholder.classList.remove('hidden');
                btn.innerText = "Mulai Kamera";
                btn.classList.remove('bg-red-50', 'text-red-600');
            }
        };

        function handleScan(id) {
            const emp = employees.find(e => e.id === id);
            if (!emp) return showToast("ID tidak dikenal!", "error");

            const today = new Date().toLocaleDateString('id-ID');
            const nowTime = new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' }).replace('.', ':');
            
            let log = logs.find(l => l.id === id && l.date === today);

            if (!log) {
                // Absen Masuk
                const isLate = nowTime > JAM_MASUK_MAKS;
                logs.unshift({
                    id: id,
                    name: emp.name,
                    date: today,
                    timeIn: nowTime,
                    timeOut: null,
                    isLate: isLate
                });
                showToast(`Halo ${emp.name}, Selamat bekerja!`, "success");
            } else if (!log.timeOut) {
                // Absen Keluar
                log.timeOut = nowTime;
                showToast(`Sampai jumpa ${emp.name}!`, "success");
            } else {
                showToast("Anda sudah absen keluar hari ini", "warning");
            }

            saveAndSync();
        }

        // --- REPORTING & EDIT ---
        function renderLogs() {
            const recentEl = document.getElementById('recent-logs');
            const adminEl = document.getElementById('full-report-list');

            // User View (Top 5)
            recentEl.innerHTML = logs.slice(0, 5).map(log => `
                <div class="bg-white p-3 rounded-2xl border border-slate-100 flex justify-between items-center shadow-sm">
                    <div class="flex items-center gap-3">
                        <div class="w-8 h-8 rounded-full bg-slate-50 flex items-center justify-center text-slate-400">
                            <i data-lucide="user" class="w-4 h-4"></i>
                        </div>
                        <div>
                            <p class="text-sm font-bold text-slate-700">${log.name}</p>
                            <p class="text-[10px] text-slate-400 uppercase font-mono">${log.id}</p>
                        </div>
                    </div>
                    <div class="text-right">
                        <p class="text-xs font-bold text-slate-600">${log.timeIn} ${log.timeOut ? ' - ' + log.timeOut : ''}</p>
                        <span class="text-[9px] px-2 py-0.5 rounded-full font-bold ${log.timeOut ? 'bg-indigo-100 text-indigo-600' : 'bg-green-100 text-green-600'}">
                            ${log.timeOut ? 'SELESAI' : 'BEKERJA'}
                        </span>
                    </div>
                </div>
            `).join('');

            // Admin View (Full with Edit)
            adminEl.innerHTML = logs.map((log, index) => {
                const totalHours = calculateHours(log.timeIn, log.timeOut);
                const isOvertime = totalHours > 8;
                
                return `
                <div class="bg-white p-4 rounded-2xl border border-slate-200 shadow-sm">
                    <div class="flex justify-between items-start mb-2">
                        <div>
                            <p class="text-xs text-slate-400">${log.date}</p>
                            <p class="font-bold text-slate-800">${log.name}</p>
                        </div>
                        <button onclick="openEdit(${index})" class="p-2 bg-slate-50 rounded-lg text-blue-500">
                            <i data-lucide="edit-3" class="w-4 h-4"></i>
                        </button>
                    </div>
                    <div class="grid grid-cols-2 gap-2 text-center">
                        <div class="bg-slate-50 p-2 rounded-xl">
                            <p class="text-[10px] text-slate-400 font-bold uppercase">Masuk</p>
                            <p class="text-sm font-bold ${log.isLate ? 'text-red-500' : 'text-slate-700'}">${log.timeIn}</p>
                            ${log.isLate ? '<span class="text-[8px] text-red-400 font-bold">TERLAMBAT</span>' : ''}
                        </div>
                        <div class="bg-slate-50 p-2 rounded-xl">
                            <p class="text-[10px] text-slate-400 font-bold uppercase">Keluar</p>
                            <p class="text-sm font-bold text-slate-700">${log.timeOut || '--:--'}</p>
                            ${isOvertime ? '<span class="text-[8px] text-amber-500 font-bold">LEMBUR</span>' : ''}
                        </div>
                    </div>
                </div>
            `}).join('');
            
            lucide.createIcons();
        }

        // Helper: Hitung Jam Kerja
        function calculateHours(start, end) {
            if (!start || !end) return 0;
            const s = start.split(':');
            const e = end.split(':');
            const diff = (parseInt(e[0])*60 + parseInt(e[1])) - (parseInt(s[0])*60 + parseInt(s[1]));
            return diff / 60;
        }

        // Edit Modal Logic
        function openEdit(index) {
            currentEditIdx = index;
            const log = logs[index];
            document.getElementById('edit-in').value = log.timeIn.replace('.', ':');
            document.getElementById('edit-out').value = log.timeOut ? log.timeOut.replace('.', ':') : '';
            document.getElementById('edit-modal').classList.remove('hidden');
        }

        function closeEditModal() {
            document.getElementById('edit-modal').classList.add('hidden');
        }

        function saveEdit() {
            const log = logs[currentEditIdx];
            log.timeIn = document.getElementById('edit-in').value;
            log.timeOut = document.getElementById('edit-out').value || null;
            log.isLate = log.timeIn > JAM_MASUK_MAKS;
            
            saveAndSync();
            closeEditModal();
            showToast("Data diperbarui", "success");
        }

        function deleteLog() {
            if(confirm('Hapus catatan ini?')) {
                logs.splice(currentEditIdx, 1);
                saveAndSync();
                closeEditModal();
                showToast("Data dihapus", "warning");
            }
        }

        function saveAndSync() {
            localStorage.setItem('absensi_logs', JSON.stringify(logs));
            renderLogs();
        }

        // Toast Notification
        function showToast(msg, type) {
            const toast = document.getElementById('toast');
            const bg = document.getElementById('toast-bg');
            const txt = document.getElementById('toast-text');
            const icon = document.getElementById('toast-icon');

            txt.innerText = msg;
            toast.classList.remove('hidden');
            
            if (type === 'success') {
                bg.className = "px-6 py-3 rounded-full shadow-2xl flex items-center gap-3 bg-emerald-600 text-white font-medium";
                icon.innerHTML = '<i data-lucide="check-circle" class="w-5 h-5"></i>';
            } else if (type === 'error') {
                bg.className = "px-6 py-3 rounded-full shadow-2xl flex items-center gap-3 bg-red-600 text-white font-medium";
                icon.innerHTML = '<i data-lucide="x-octagon" class="w-5 h-5"></i>';
            } else {
                bg.className = "px-6 py-3 rounded-full shadow-2xl flex items-center gap-3 bg-amber-500 text-white font-medium";
                icon.innerHTML = '<i data-lucide="alert-triangle" class="w-5 h-5"></i>';
            }

            lucide.createIcons();
            setTimeout(() => toast.classList.add('hidden'), 3000);
        }

        function downloadCSV() {
            let csv = "ID,Nama,Tanggal,Masuk,Keluar,Terlambat,Status\n";
            logs.forEach(l => {
                const hours = calculateHours(l.timeIn, l.timeOut);
                const status = hours > 8 ? "Lembur" : (l.timeOut ? "Hadir" : "Aktif");
                csv += `${l.id},${l.name},${l.date},${l.timeIn},${l.timeOut || '-'},${l.isLate ? 'YA' : 'TIDAK'},${status}\n`;
            });
            
            const blob = new Blob([csv], { type: 'text/csv' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `Laporan_Absensi_${new Date().toLocaleDateString()}.csv`;
            a.click();
        }
